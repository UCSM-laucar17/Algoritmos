static double potencia(double x, int y){ 
    if(y == 0){                              //O(1) 
       return 1.0;                           //O(1) 
    } 
   if(y % 2 == 1){                           //O(1)+O(1) 
       return x * potencia(x, y - 1);        //O(1)+O(1)+T(y-1)+O(1) 
   }else{ 
       double t = potencia(x, y / 2);        //O(1)+T(y/2)+O(1)
       return t * t;                         //O(1)+O(1) 
   } 
}
